# read csv into R - full

	skip = 0,					# number of lines to skip
	stringsAsFactors = FALSE,	#

	sep = ",",					# field separator
	quote = "\"", 				# quoting characters
	strip.white = TRUE,			# strip whitespace?
	na.strings=c("", "NA"), 	# what to do with NAs (captures blanks and "NA")
	)
